#include<bits/stdc++.h>
using namespace std;

int arr[100000];

int main()
{
    int n,maxi=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        cin>>arr[i];
        if(maxi<arr[i]){
            maxi=arr[i];
        }
    }
    sort(arr,arr+n);
    for(int i=n-2;i>=0;i--){
        maxi=maxi-arr[i];
    }
    cout<<maxi<<endl;;



}
